STEP 1 LOCAL LAPTOP (Has Internet)
Auto-Download All 4 Patches
This script runs on your LOCAL work laptop (the one with internet). It downloads Chrome, Edge, Defender, and the CURRENT MONTH's Windows Server 2019 Cumulative Update from the Microsoft Update Catalog. No guessing KB numbers — it finds the right one automatically.
Paste this into PowerShell (Run as Administrator) on your local laptop. It creates a dated folder and downloads everything.
# ================================================ # SAP NS2 IL5 — Auto Patch Downloader # Run on LOCAL LAPTOP with Internet — PowerShell as Admin # Downloads: Chrome, Edge, Defender, Windows KB (Server 2019) # Finds CURRENT month KB from Microsoft Update Catalog # ================================================ $Month = "apr-2026" $MonthNum = "04" $Year = "2026" $DownDir = "C:\Temp\Patches\$Month" Write-Host "================================================" -ForegroundColor Cyan Write-Host " SAP NS2 IL5 — Patch Downloader — $Month" -ForegroundColor Green Write-Host "================================================" -ForegroundColor Cyan # Create download folder New-Item -ItemType Directory -Path $DownDir -Force | Out-Null Write-Host "Download folder: $DownDir" -ForegroundColor Green # ------- 1. CHROME ENTERPRISE ------- Write-Host "`n[1/4] Downloading Chrome Enterprise..." -ForegroundColor Yellow $chromeUrl = "https://dl.google.com/chrome/install/googlechromestandaloneenterprise64.msi" $chromeDst = "$DownDir\GoogleChromeStandaloneEnterprise64.msi" try { Invoke-WebRequest -Uri $chromeUrl -OutFile $chromeDst -UseBasicParsing Write-Host " Chrome: $([math]::Round((Get-Item $chromeDst).Length/1MB))MB" -ForegroundColor Green } catch { Write-Host " Chrome FAILED: $_" -ForegroundColor Red } # ------- 2. EDGE ENTERPRISE ------- Write-Host "`n[2/4] Downloading Edge Enterprise..." -ForegroundColor Yellow $edgeUrl = "https://msedge.sf.dl.delivery.mp.microsoft.com/filestreamingservice/files/latest/MicrosoftEdgeEnterpriseX64.msi" $edgeDst = "$DownDir\MicrosoftEdgeEnterpriseX64.msi" try { Invoke-WebRequest -Uri $edgeUrl -OutFile $edgeDst -UseBasicParsing Write-Host " Edge: $([math]::Round((Get-Item $edgeDst).Length/1MB))MB" -ForegroundColor Green } catch { Write-Host " Edge FAILED: $_" -ForegroundColor Red } # ------- 3. DEFENDER SIGNATURES ------- Write-Host "`n[3/4] Downloading Defender Signatures..." -ForegroundColor Yellow $defUrl = "https://go.microsoft.com/fwlink/?LinkID=121721&arch=x64" $defDst = "$DownDir\mpam-fe.exe" try { Invoke-WebRequest -Uri $defUrl -OutFile $defDst -UseBasicParsing Write-Host " Defender: $([math]::Round((Get-Item $defDst).Length/1MB))MB" -ForegroundColor Green } catch { Write-Host " Defender FAILED: $_" -ForegroundColor Red } # ------- 4. WINDOWS KB — FROM MICROSOFT UPDATE CATALOG ------- Write-Host "`n[4/4] Searching Microsoft Update Catalog for Server 2019 CU..." -ForegroundColor Yellow Write-Host " Searching: $Year-$MonthNum Cumulative Update for Windows Server 2019" -ForegroundColor Gray $SearchTerm = "$Year-$MonthNum Cumulative Update for Windows Server 2019 for x64" $CatalogURL = "https://www.catalog.update.microsoft.com/Search.aspx?q=$([uri]::EscapeDataString($SearchTerm))" try { $SearchPage = Invoke-WebRequest -Uri $CatalogURL -UseBasicParsing -ErrorAction Stop # Extract update IDs and titles from results $IDMatches = [regex]::Matches($SearchPage.Content, 'goToDetails\("([a-f0-9\-]+)"\)') $TitleMatches = [regex]::Matches($SearchPage.Content, 'class="resulttitle"[^>]*>\s*<span[^>]*>([^<]+)') # Find the right CU — must say "Cumulative Update" and "Server 2019" and NOT "Dynamic" $UpdateID = $null $KBTitle = "" $KBNumber = "" for ($i = 0; $i -lt $IDMatches.Count; $i++) { $title = if ($i -lt $TitleMatches.Count) { $TitleMatches[$i].Groups[1].Value } else { "" } if ($title -match "Cumulative Update" -and $title -match "Server 2019" -and $title -notmatch "Dynamic") { $UpdateID = $IDMatches[$i].Groups[1].Value $KBTitle = $title if ($title -match "(KB\d+)") { $KBNumber = $Matches[1] } break } } # Fallback: try first result if specific match not found if (-not $UpdateID -and $IDMatches.Count -gt 0) { $UpdateID = $IDMatches[0].Groups[1].Value if ($TitleMatches.Count -gt 0) { $KBTitle = $TitleMatches[0].Groups[1].Value } if ($KBTitle -match "(KB\d+)") { $KBNumber = $Matches[1] } } if ($UpdateID) { Write-Host " Found: $KBTitle" -ForegroundColor Green Write-Host " KB: $KBNumber | ID: $UpdateID" -ForegroundColor Gray # Get download URL from catalog $PostBody = @{ updateIDs = "[{`"size`":0,`"uidInfo`":`"$UpdateID`",`"updateID`":`"$UpdateID`"}]" } $DownPage = Invoke-WebRequest -Uri "https://www.catalog.update.microsoft.com/DownloadDialog.aspx" -Method Post -Body $PostBody -UseBasicParsing $DownURL = [regex]::Match($DownPage.Content, "(http[s]?://[^'`"]*\.msu)").Value if ($DownURL) { Write-Host " Downloading $KBNumber (~800MB, be patient)..." -ForegroundColor Yellow $kbDst = "$DownDir\kb-$Month.msu" Invoke-WebRequest -Uri $DownURL -OutFile $kbDst -UseBasicParsing Write-Host " $KBNumber : $([math]::Round((Get-Item $kbDst).Length/1MB))MB — SAVED AS kb-$Month.msu" -ForegroundColor Green } else { Write-Host " Could not extract download URL. Opening catalog in browser..." -ForegroundColor Red Start-Process $CatalogURL Write-Host " Download the .msu manually and rename to: kb-$Month.msu" -ForegroundColor Yellow } } else { Write-Host " No CU found for $Year-$MonthNum. Opening catalog..." -ForegroundColor Red Start-Process $CatalogURL Write-Host " Download the .msu manually and rename to: kb-$Month.msu" -ForegroundColor Yellow } } catch { Write-Host " Catalog search failed: $_" -ForegroundColor Red Write-Host " Opening catalog in browser as fallback..." -ForegroundColor Yellow Start-Process $CatalogURL Write-Host " Search for: $SearchTerm" -ForegroundColor Yellow Write-Host " Download the .msu and save as: $DownDir\kb-$Month.msu" -ForegroundColor Yellow } # ------- FINAL REPORT ------- Write-Host "`n================================================" -ForegroundColor Cyan Write-Host " DOWNLOAD REPORT — $Month" -ForegroundColor Green Write-Host "================================================" -ForegroundColor Cyan Get-ChildItem $DownDir | Format-Table Name, @{N="Size(MB)";E={[math]::Round($_.Length/1MB)}}, LastWriteTime -AutoSize Write-Host "Total files: $((Get-ChildItem $DownDir).Count)" -ForegroundColor Green Write-Host "Location: $DownDir" -ForegroundColor Green Write-Host "`nNEXT: Open WinSCP and upload these 4 files to IL5 SFTP" -ForegroundColor Yellow

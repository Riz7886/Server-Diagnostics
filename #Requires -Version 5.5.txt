#Requires -Version 5.1
[CmdletBinding()]
param(
    [string]$SignedCert      = 'C:\temp\script\opt.cer',
    [string]$WorkDir         = 'C:\temp\script',
    [string]$TargetServer    = '10.168.0.32',
    [string]$TargetShare     = 'e$\CERTS',
    [string]$ProdKeystore    = 's4pceotcac.p12',
    [string]$NewKeystoreName = 's4pceotcac_new.p12',
    [string]$PwFile          = 'pw.xml',
    [string]$Alias           = 's4hpce',
    [string]$ServiceName     = 'OpenText Core Archive Connector',
    [string]$KeytoolPath,
    [string]$Password
)

$ErrorActionPreference = 'Stop'

function Say  ($m,$c='Cyan')  { Write-Host "[*] $m"   -ForegroundColor $c }
function Ok   ($m)            { Write-Host "[ok] $m"  -ForegroundColor Green }
function Warn ($m)            { Write-Host "[!!] $m"  -ForegroundColor Yellow }
function Die  ($m)            { Write-Host "[xx] $m"  -ForegroundColor Red; exit 1 }

function Find-Keytool {
    param([string]$Hint)
    if ($Hint -and (Test-Path $Hint)) { return $Hint }
    $cmd = Get-Command keytool.exe -ErrorAction SilentlyContinue
    if ($cmd) { return $cmd.Path }
    $paths = @(
        'C:\Program Files\Java\jre*\bin\keytool.exe',
        'C:\Program Files\Java\jdk*\bin\keytool.exe',
        'C:\Program Files\Eclipse Adoptium\*\bin\keytool.exe',
        'C:\Program Files\OpenJDK\*\bin\keytool.exe',
        'C:\Program Files\Amazon Corretto\*\bin\keytool.exe',
        'C:\Program Files (x86)\Java\*\bin\keytool.exe',
        'C:\Java\*\bin\keytool.exe'
    )
    foreach ($p in $paths) {
        $f = Get-ChildItem -Path $p -ErrorAction SilentlyContinue | Select-Object -First 1
        if ($f) { return $f.FullName }
    }
    return $null
}

function Find-PwXml {
    $direct = @(
        (Join-Path $WorkDir $PwFile),
        "C:\temp\script\$PwFile",
        "C:\temp\script\work\$PwFile",
        "C:\temp\$PwFile"
    )
    foreach ($p in $direct) { if (Test-Path $p) { return $p } }
    $hit = Get-ChildItem -Path C:\temp -Recurse -Filter $PwFile -Force -ErrorAction SilentlyContinue | Select-Object -First 1
    if ($hit) { return $hit.FullName }
    return $null
}

Say "Preflight"
if (-not (Test-Path $SignedCert)) { Die "Signed cert not found: $SignedCert" }
Ok "Signed cert: $SignedCert"

$keystoreLocal = Join-Path $WorkDir $NewKeystoreName
if (-not (Test-Path $keystoreLocal)) { Die "Local keystore missing at $keystoreLocal" }
Ok "Keystore: $keystoreLocal"

Say "Verify cert contents (is this really the fresh signed cert?)"
try {
    $certObj = New-Object System.Security.Cryptography.X509Certificates.X509Certificate2 $SignedCert
    Write-Host "  Subject    : $($certObj.Subject)"
    Write-Host "  Issuer     : $($certObj.Issuer)"
    Write-Host "  NotBefore  : $($certObj.NotBefore)"
    Write-Host "  NotAfter   : $($certObj.NotAfter)"
    Write-Host "  Thumbprint : $($certObj.Thumbprint)"
    $now = Get-Date
    if ($now -gt $certObj.NotAfter) { Die "Cert already expired on $($certObj.NotAfter) - Rhonda sent wrong file" }
    if ($now -lt $certObj.NotBefore) { Die "Cert not yet valid (starts $($certObj.NotBefore))" }
    $ageDays = ($now - $certObj.NotBefore).TotalDays
    if ($ageDays -gt 30) {
        Warn "Cert's NotBefore is $([int]$ageDays) days ago - could be the OLD 2024 cert. Double-check before proceeding."
        Start-Sleep -Seconds 5
    } else {
        Ok "Cert looks fresh (issued $([int]$ageDays) days ago)"
    }
    $daysLeft = ($certObj.NotAfter - $now).TotalDays
    Ok "Valid for $([int]$daysLeft) more days"
} catch {
    Die "Could not parse cert at $SignedCert : $($_.Exception.Message)"
}

$p1 = $null
if ($Password) {
    $p1 = $Password
    Ok "Using -Password parameter"
} else {
    $pwStore = Find-PwXml
    if ($pwStore) {
        Ok "pw.xml found: $pwStore"
        try {
            $pw = Import-Clixml -Path $pwStore
            $p1 = [System.Net.NetworkCredential]::new('', $pw).Password
        } catch {
            Warn "pw.xml unreadable: $($_.Exception.Message)"
            $p1 = $null
        }
    }
}
if (-not $p1) {
    Die "No password source. Re-run: .\renew-prod.ps1 -Password 'yourPhase1Password'"
}

$kt = Find-Keytool -Hint $KeytoolPath
if (-not $kt) { Die "keytool.exe not found. Pass -KeytoolPath 'C:\path\to\keytool.exe'." }
Ok "keytool: $kt"

$remoteShare = "\\$TargetServer\$TargetShare"
if (-not (Test-Path $remoteShare)) { Die "SMB share not reachable: $remoteShare" }
Ok "Remote share reachable: $remoteShare"

Say "Import signed cert into local keystore"
$impArgs = @(
    '-importcert','-trustcacerts','-noprompt',
    '-alias',$Alias,
    '-file',$SignedCert,
    '-keystore',$keystoreLocal,
    '-storetype','PKCS12',
    '-storepass',$p1
)
& $kt @impArgs
if ($LASTEXITCODE -ne 0) { Die "keytool -importcert failed: exit $LASTEXITCODE (wrong password or wrong alias)" }
Ok "Imported signed cert."

Say "Verify keystore contents"
& $kt -list -v -alias $Alias -keystore $keystoreLocal -storetype PKCS12 -storepass $p1
if ($LASTEXITCODE -ne 0) { Die "keytool -list failed after import" }

Say "Back up existing production keystore on target"
$prodPath   = Join-Path $remoteShare $ProdKeystore
$stamp      = Get-Date -Format 'yyyyMMddHHmmss'
$backupPath = Join-Path $remoteShare "$ProdKeystore.bak-$stamp"
if (Test-Path $prodPath) {
    Copy-Item $prodPath $backupPath -Force
    Ok "Prod backup: $backupPath"
} else {
    Warn "No existing prod keystore at $prodPath"
}

Say "Push new keystore to production"
Copy-Item $keystoreLocal $prodPath -Force
Ok "Prod keystore updated: $prodPath"

Say "Restart service on $TargetServer"
$restarted = $false
try {
    $svc = Get-Service -ComputerName $TargetServer -Name $ServiceName -ErrorAction Stop
    Restart-Service -InputObject $svc -Force -ErrorAction Stop
    Start-Sleep -Seconds 3
    $svc = Get-Service -ComputerName $TargetServer -Name $ServiceName
    Ok "Method 1 (RPC) succeeded. Status: $($svc.Status)"
    $restarted = $true
} catch {
    Warn "Method 1 (RPC) failed: $($_.Exception.Message)"
}

if (-not $restarted) {
    try {
        & sc.exe "\\$TargetServer" stop $ServiceName | Out-Host
        Start-Sleep -Seconds 5
        & sc.exe "\\$TargetServer" start $ServiceName | Out-Host
        Start-Sleep -Seconds 3
        & sc.exe "\\$TargetServer" query $ServiceName | Out-Host
        Ok "Method 2 (sc.exe) completed."
        $restarted = $true
    } catch {
        Warn "Method 2 (sc.exe) failed: $($_.Exception.Message)"
    }
}

if (-not $restarted) {
    try {
        $q = "SELECT * FROM Win32_Service WHERE Name='$ServiceName'"
        Invoke-CimMethod -ComputerName $TargetServer -Query $q -MethodName StopService -ErrorAction Stop | Out-Null
        Start-Sleep -Seconds 5
        Invoke-CimMethod -ComputerName $TargetServer -Query $q -MethodName StartService -ErrorAction Stop | Out-Null
        Start-Sleep -Seconds 3
        $cim = Get-CimInstance -ComputerName $TargetServer -Query $q
        Ok "Method 3 (WMI) succeeded. State: $($cim.State)"
        $restarted = $true
    } catch {
        Warn "Method 3 (WMI) failed: $($_.Exception.Message)"
    }
}

if (-not $restarted) {
    Die "All restart methods failed. Keystore swapped but service holds old cert. Ask SAP admin to restart '$ServiceName' on $TargetServer."
}

Write-Host ''
Ok "DONE. Cert renewed end-to-end on $TargetServer."
Ok "  Prod keystore : $prodPath"
Ok "  Backup        : $backupPath"
Ok "  Service       : $ServiceName restarted"

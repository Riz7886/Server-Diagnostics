📋 How to Use Every Month

🔴 Run on CLIENT SERVER via Fleet Manager RDP — NOT on the Jumphost!
  1. Connect to server via Fleet Manager → Remote Desktop
  2. Open Notepad on the client server
  3. Click the green Copy button for your server below
  4. Paste into Notepad → File → Save As → C:\patch.ps1
  5. Open PowerShell as Administrator
  6. Type C:\patch.ps1 → press Enter
  7. Wait for ALL PATCHES DONE! message
⚠️ Script uses msiexec /REINSTALL so it FORCES install even if Chrome/Edge already exist on the server. This is the correct way.

🖥️ SCRIPT A — Support002 Sidecar Server

Bucket: ns2-il5-s4-support0002-sc-ssm-patching — Path: patches/windows_patch/ — NO date folder

⚠️ If KB download fails from S3 — download from AWS Console browser → it saves to Downloads folder → script moves it automatically
# ================================================
# IL5 PATCH SCRIPT — SUPPORT002 SIDECAR SERVER
# March 2026 — Run as Administrator
# ================================================

# Comment out if stuck on this line:
# Import-Module AWSPowerShell

# --- VARIABLES — change month every month ---
$Bucket  = "ns2-il5-s4-support0002-sc-ssm-patching"
$Prefix  = "patches/windows_patch"
$Folder  = "C:\Temp\Patches\mar-2026"
$Region  = "us-gov-west-1"

# --- CREATE FOLDER AUTOMATICALLY ---
New-Item -ItemType Directory -Path $Folder -Force | Out-Null
Write-Host "Folder created: $Folder" -ForegroundColor Green

# ================================================
# DOWNLOAD ALL 4 PATCHES
# ================================================

Write-Host "[1/4] Downloading Chrome..." -ForegroundColor Yellow
Read-S3Object -BucketName $Bucket -Key "$Prefix/GoogleChromeStandaloneEnterprise64.msi" -File "$Folder\chrome.msi" -Region $Region
Write-Host "  Chrome downloaded!" -ForegroundColor Green

Write-Host "[2/4] Downloading Edge..." -ForegroundColor Yellow
Read-S3Object -BucketName $Bucket -Key "$Prefix/MicrosoftEdgeEnterpriseX64.msi" -File "$Folder\edge.msi" -Region $Region
Write-Host "  Edge downloaded!" -ForegroundColor Green

Write-Host "[3/4] Downloading Defender..." -ForegroundColor Yellow
Read-S3Object -BucketName $Bucket -Key "$Prefix/mpam-fe.exe" -File "$Folder\defender.exe" -Region $Region
Write-Host "  Defender downloaded!" -ForegroundColor Green

Write-Host "[4/4] Downloading Windows KB5078752..." -ForegroundColor Yellow
Read-S3Object -BucketName $Bucket -Key "$Prefix/windows10.0-kb5078752-x64_d19dce98bd2132ac8dcf4ae644ff36ec1ac7a754.msu" -File "$Folder\kb.msu" -Region $Region

# If KB download failed from S3 — move from Downloads folder instead
if (-not (Test-Path "$Folder\kb.msu") -or (Get-Item "$Folder\kb.msu").Length -lt 1000) {
    Write-Host "  KB not in S3 — checking Downloads folder..." -ForegroundColor Yellow
    $dl = Get-ChildItem "C:\Users\Administrator\Downloads\windows10.0-kb5078752*.msu" -ErrorAction SilentlyContinue | Select-Object -First 1
    if ($dl) {
        Move-Item $dl.FullName "$Folder\kb.msu" -Force
        Write-Host "  KB moved from Downloads!" -ForegroundColor Green
    } else {
        Write-Host "  KB NOT FOUND — download manually from AWS Console S3 browser" -ForegroundColor Red
    }
} else {
    Write-Host "  Windows KB downloaded!" -ForegroundColor Green
}

Write-Host "ALL FILES DOWNLOADED!" -ForegroundColor Cyan

# ================================================
# INSTALL ALL 4 PATCHES — FORCE REINSTALL
# ================================================

Write-Host "Installing Chrome (force reinstall)..." -ForegroundColor Yellow
Start-Process "msiexec.exe" -ArgumentList "/i `"$Folder\chrome.msi`" /quiet /norestart REINSTALL=ALL REINSTALLMODE=vomus" -Wait
Write-Host "  Chrome installed!" -ForegroundColor Green

Write-Host "Installing Edge (force reinstall)..." -ForegroundColor Yellow
Start-Process "msiexec.exe" -ArgumentList "/i `"$Folder\edge.msi`" /quiet /norestart REINSTALL=ALL REINSTALLMODE=vomus" -Wait
Write-Host "  Edge installed!" -ForegroundColor Green

Write-Host "Installing Defender..." -ForegroundColor Yellow
Start-Process -FilePath "$Folder\defender.exe" -Wait
Write-Host "  Defender installed!" -ForegroundColor Green

Write-Host "Installing Windows KB5078752..." -ForegroundColor Yellow
Start-Process -FilePath "wusa.exe" -ArgumentList "$Folder\kb.msu /quiet /norestart" -Wait
Write-Host "  Windows KB installed!" -ForegroundColor Green

Write-Host "================================================" -ForegroundColor Cyan
Write-Host "ALL PATCHES DONE!" -ForegroundColor Green
Write-Host "================================================" -ForegroundColor Cyan

# ================================================
# VERIFY
# ================================================
Write-Host "Chrome version:" -ForegroundColor Yellow
(Get-Item "C:\Program Files\Google\Chrome\Application\chrome.exe").VersionInfo.FileVersion

Write-Host "Edge version:" -ForegroundColor Yellow
(Get-Item "C:\Program Files (x86)\Microsoft\Edge\Application\msedge.exe").VersionInfo.FileVersion

Write-Host "Defender update date:" -ForegroundColor Yellow
Get-MpComputerStatus | Select-Object AntivirusSignatureLastUpdated, AntivirusSignatureVersion

Write-Host "KB5078752 check:" -ForegroundColor Yellow
Get-HotFix | Where-Object {$_.HotFixID -eq "KB5078752"}

Write-Host "  Defender downloaded!" -ForegroundColor Green

# Download Windows KB5078752
Write-Host "[4/4] Downloading KB5078752..." -ForegroundColor Yellow
Read-S3Object -BucketName "ns2-il5-s4-support0002-sc-ssm-patching" -Key "patches/windows_patch/windows10.0-kb5078752-x64_d19dce98bd2132ac8dcf4ae644ff36ec1ac7a754.msu" -File "C:\Temp\Patches\mar-2026\kb.msu" -Region "us-gov-west-1"

Write-Host "  Chrome installed!" -ForegroundColor Green

# Install Edge
Write-Host "Installing Edge..." -ForegroundColor Yellow
Start-Process "msiexec.exe" -ArgumentList "/i `"C:\Temp\Patches\mar-2026\edge.msi`" /quiet /norestart REINSTALL=ALL REINSTALLMODE=vomus" -Wait
Write-Host "  Edge installed!" -ForegroundColor Green

# Install Defender
Write-Host "Installing Defender..." -ForegroundColor Yellow
Start-Process "msiexec.exe" -ArgumentList "/i `"C:\Temp\Patches\mar-2026\defender.exe`" /quiet /norestart" -Wait
Write-Host "  Defender installed!" -ForegroundColor Green

# Install Windows KB
Write-Host "Installing KB5078752..." -ForegroundColor Yellow
Start-Process "wusa.exe" -ArgumentList "C:\Temp\Patches\mar-2026\kb.msu /quiet /norestart" -Wait
Write-Host "  KB installed!" -ForegroundColor Green

Write-Host "================================================" -ForegroundColor Cyan
Write-Host "ALL PATCHES DONE!" -ForegroundColor Green
Write-Host "================================================" -ForegroundColor Cyan

# Verify
Write-Host "Chrome:" -ForegroundColor Yellow
(Get-Item "C:\Program Files\Google\Chrome\Application\chrome.exe").VersionInfo.FileVersion

Write-Host "Edge:" -ForegroundColor Yellow
(Get-Item "C:\Program Files (x86)\Microsoft\Edge\Application\msedge.exe").VersionInfo.FileVersion

Write-Host "Defender:" -ForegroundColor Yellow
Get-MpComputerStatus | Select-Object AntivirusSignatureLastUpdated, AntivirusSignatureVersion

Write-Host "KB check:" -ForegroundColor Yellow
Get-HotFix | Where-Object {$_.HotFixID -eq "KB5078752"}
